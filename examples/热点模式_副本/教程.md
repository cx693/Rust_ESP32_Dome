## 📝 核心代码讲解

### 一、AP 热点配置

```rust
const AP_SSID: &str = "ESP32-AP";
const AP_PASSWORD: &str = "12345678";

// 配置 AP 模式
let ap_config = Config::AccessPoint(
    AccessPointConfig::default()
        .with_ssid(AP_SSID)
        .with_password(String::from(AP_PASSWORD))
        .with_auth_method(AuthenticationMethod::Wpa2Personal),  // WPA2 加密
);

// 创建 WiFi 驱动
let (controller, interfaces) = esp_radio::wifi::new(
    peripherals.WIFI,
    ControllerConfig::default().with_initial_config(ap_config),
).unwrap();

// 静态 IP 配置
let (stack, runner) = embassy_net::new(
    interfaces.access_point,
    embassy_net::Config::ipv4_static(StaticConfigV4 {
        address: Ipv4Cidr::new(Ipv4Address::new(192, 168, 4, 1), 24),
        gateway: None,
        dns_servers: Default::default(),
    }),
    mk_static!(StackResources<3>, StackResources::<3>::new()),
    seed,
);
```

### 二、自定义 DHCP 服务器

```rust
#[embassy_executor::task]
async fn dhcp_task(stack: Stack<'static>) {
    let mut socket = UdpSocket::new(stack, ...);
    socket.bind(67).unwrap();  // DHCP 服务器端口

    loop {
        // 接收 DHCP 请求
        let (n, _src) = socket.recv_from(&mut recv_data).await.unwrap();
        let pkt = &recv_data[..n];

        // 验证 DHCP 魔数
        if pkt[236..240] != [99, 130, 83, 99] { continue; }

        // 提取消息类型
        let msg_type = find_dhcp_option(pkt, 53).unwrap()[0];

        match msg_type {
            1 => {
                // DHCP Discover → 分配 IP → 发送 Offer
                let ip_octet = allocate_ip(client_mac);
                let offer_ip = [192, 168, 4, ip_octet];
                let len = build_dhcp_response(&mut resp, 2, &xid, &client_mac, &offer_ip);
                socket.send_to(&resp[..len], broadcast).await.ok();
            }
            3 => {
                // DHCP Request → 确认分配 → 发送 ACK
                let ack_ip = [192, 168, 4, ip_octet];
                let len = build_dhcp_response(&mut resp, 5, &xid, &client_mac, &ack_ip);
                socket.send_to(&resp[..len], broadcast).await.ok();
            }
            _ => {}
        }
    }
}
```

**DHCP 流程：**
```
客户端                    ESP32 DHCP Server
  │                           │
  │──── DHCP Discover ───────→│  (广播: "有人吗?")
  │                           │
  │←──── DHCP Offer ─────────│  (单播: "给你 192.168.4.10")
  │                           │
  │──── DHCP Request ────────→│  (广播: "我要 192.168.4.10")
  │                           │
  │←──── DHCP ACK ───────────│  (单播: "确认!")
  │                           │
  客户端获得 IP: 192.168.4.10
```

### 三、Web 仪表盘

```rust
// HTTP 路由
match (method, path) {
    ("GET", "/") => {
        // 返回 HTML 仪表盘页面
        respond(&mut socket, "200 OK", "text/html", HTML.as_bytes()).await;
    }
    ("GET", "/api/clients") => {
        // 返回已连接设备 JSON
        let json = clients_json();
        respond(&mut socket, "200 OK", "application/json", json.as_bytes()).await;
    }
}
```
